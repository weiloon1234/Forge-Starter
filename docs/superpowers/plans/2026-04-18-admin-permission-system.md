# Admin Permission System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a per-admin permission system with tier-based hierarchy enforcement, driven by a single Rust `Permission` enum that feeds route gates, DB storage, validation, OpenAPI, and TypeScript.

**Architecture:** Bypass-by-tier (SuperAdmin + Developer get `abilities = ["*"]` via middleware); plain Admin governed by a JSONB `permissions` column. Service-layer guards enforce the hierarchy matrix (Developer > SuperAdmin > Admin; nobody edits peers or self). Permissions declared once in `src/ids/permissions.rs` via `#[derive(forge::AppEnum, ts_rs::TS)]`; TypeScript types auto-generated by `make types`.

**Tech Stack:** Rust (Forge framework, axum), PostgreSQL (JSONB), React (Vite), ts-rs.

**Spec:** `docs/superpowers/specs/2026-04-18-admin-permission-system-design.md`

```rust
pub struct Admin {
    pub id: ModelId<Self>,
    pub username: String,
    pub email: String,
    pub name: String,

    #[forge(write_mutator = "hash_password")]
    pub password_hash: String,
    pub permissions: Vec<String>,
    pub locale: String,
    pub created_at: DateTime,
    pub updated_at: DateTime,
}

impl Admin {
    async fn hash_password(ctx: &ModelHookContext<'_>, value: String) -> Result<String> {
        ctx.app().hash()?.hash(&value)
    }

    pub fn is_bypass(&self) -> bool {
        matches!(self.admin_type, AdminType::SuperAdmin | AdminType::Developer)
    }

    pub fn effective_abilities(&self) -> Vec<String> {
        if self.is_bypass() {
            vec!["*".into()]
        } else {
            self.permissions.clone()
        }
    }
}

impl HasToken for Admin {
    fn token_actor_id(&self) -> String {
        self.id.to_string()
    }
}

impl Authenticatable for Admin {
    fn guard() -> GuardId {
        Guard::Admin.into()
    }
}
```

```rust
// src/http/middleware/admin_abilities.rs
use crate::domain::models::Admin;
use axum::{extract::Request, middleware::Next, response::Response};
use forge::prelude::*;

/// After admin auth loads the Admin row into request extensions, rewrite the
/// `AuthUser<Admin>.abilities` vector based on admin_type:
///   - SuperAdmin / Developer -> ["*"] (bypass)
///   - Admin                  -> value from admins.permissions column
pub async fn admin_abilities_middleware(
    mut req: Request,
    next: Next,
) -> std::result::Result<Response, AppError> {
    if let Some(auth) = req.extensions_mut().get_mut::<AuthUser<Admin>>() {
        auth.abilities = auth.user.effective_abilities();
    }
    Ok(next.run(req).await)
}

#[cfg(test)]
```

```rust
// src/domain/services/admin_service.rs
use crate::domain::enums::AdminType;
use crate::domain::models::Admin;
use crate::ids::permissions::Permission;
use forge::prelude::*;

/// Strict ordering: Developer > SuperAdmin > Admin.
/// An actor may act on a target only when `rank(actor) > rank(target)`.
fn rank(t: &AdminType) -> u8 {
    match t {
        AdminType::Developer => 3,
        AdminType::SuperAdmin => 2,
        AdminType::Admin => 1,
    }
}

pub fn ensure_not_self(actor: &Admin, target_id: &ModelId<Admin>, i18n: &I18n) -> Result<()> {
    if actor.id == *target_id {
        return Err(Error::forbidden(forge::t!(i18n, "admin.errors.cannot_edit_self")));
    }
    Ok(())
}

pub fn ensure_can_manage(actor: &Admin, target: &Admin, i18n: &I18n) -> Result<()> {
    if rank(&actor.admin_type) <= rank(&target.admin_type) {
        return Err(Error::forbidden(forge::t!(
            i18n,
            "admin.errors.cannot_manage_tier"
        )));
    }
    Ok(())
}

pub fn ensure_admin_type_allowed(
    actor: &Admin,
    requested: &AdminType,
    i18n: &I18n,
) -> Result<()> {
    // Developer is never UI-settable.
    if matches!(requested, AdminType::Developer) {
        return Err(Error::forbidden(forge::t!(
            i18n,
            "admin.errors.invalid_admin_type"
        )));
    }
    if rank(&actor.admin_type) <= rank(requested) {
        return Err(Error::forbidden(forge::t!(
            i18n,
            "admin.errors.invalid_admin_type"
        )));
    }
    Ok(())
}

pub fn ensure_can_grant(
    actor: &Admin,
    requested: &[String],
    i18n: &I18n,
) -> Result<()> {
    if actor.is_bypass() {
        return Ok(());
    }
    for perm in requested {
        // Permission matcher: does actor hold (or implicitly hold) `perm`?
        if !forge::authz::has_permission(&actor.permissions, perm.as_str()) {
            return Err(Error::forbidden(forge::t!(
                i18n,
                "admin.errors.cannot_grant_permission",
                permission = perm
            )));
        }
    }
    Ok(())
}
```

```rust
pub async fn create(
    app: &AppContext,
    actor: &Admin,
    req: &CreateAdminRequest,
    i18n: &I18n,
) -> Result<Admin> {
    ensure_admin_type_allowed(actor, &req.admin_type, i18n)?;
    ensure_can_grant(actor, &req.permissions, i18n)?;

    let db = app.database()?;
    let admin = Admin::new_builder()
        .username(&req.username)
        .email(&req.email)
        .name(&req.name)
        .admin_type(req.admin_type.clone())
        .password_hash(&req.password)   // hashed by the write_mutator
        .permissions(req.permissions.clone())
        .locale(&req.locale)
        .create(&*db)
        .await?;

    ensure_can_manage(actor, &target, i18n)?;

    if let Some(t) = &req.admin_type {
        ensure_admin_type_allowed(actor, t, i18n)?;
    }
    if let Some(p) = &req.permissions {
        ensure_can_grant(actor, p, i18n)?;
    }

    let db = app.database()?;
    let mut builder = Admin::update_builder(target);
    if let Some(v) = &req.name { builder = builder.name(v); }
    if let Some(v) = &req.email { builder = builder.email(v); }
    if let Some(v) = &req.password { builder = builder.password_hash(v); }
    if let Some(v) = &req.permissions { builder = builder.permissions(v.clone()); }
    if let Some(v) = &req.admin_type { builder = builder.admin_type(v.clone()); }
    if let Some(v) = &req.locale { builder = builder.locale(v); }
    builder.save(&*db).await
}
```

```rust
        let value = match ctx.string_value() {

        let known: Vec<&'static str> = Permission::options()
            .iter()

        }
    }
}
```

```ts
// frontend/admin/src/hooks/usePermission.ts
import type { Permission } from "@shared/types/Permission";
import { useAuth } from "@shared/auth";

export function hasPermission(abilities: string[], required: Permission | string): boolean {
  if (abilities.includes("*")) return true;
  if (abilities.includes(required)) return true;

  const [res, act] = required.split(".");
  if (res && abilities.includes(`${res}.*`)) return true;
  if (act === "read" && res && abilities.includes(`${res}.manage`)) return true;

  return false;
}

export function usePermission(required: Permission | string): boolean {
  const { user } = useAuth();
  return hasPermission(user?.abilities ?? [], required);
}
```

```tsx
// frontend/admin/src/components/PermissionMatrix.tsx
import { useEffect, useMemo, useState } from "react";
import { useTranslation } from "react-i18next";
import { type Permission, PermissionValues } from "@shared/types/Permission";
import { api } from "@shared/api";
import { Radio, RadioGroup } from "@shared/components";

type Level = "none" | "read" | "manage";

interface Grantable { permission: string; grantable: boolean; }

interface Props {
  value: Permission[];
  onChange: (next: Permission[]) => void;
}

function groupModules(all: Permission[]) {
  const modules = new Map<string, { read?: Permission; manage?: Permission }>();
  for (const p of all) {
    const [res, act] = p.split(".") as [string, "read" | "manage"];
    const entry = modules.get(res) ?? {};
    entry[act] = p;
    modules.set(res, entry);
  }
  return modules;
}

function currentLevel(value: Permission[], module: { read?: Permission; manage?: Permission }): Level {
  if (module.manage && value.includes(module.manage)) return "manage";
  if (module.read && value.includes(module.read)) return "read";
  return "none";
}

export function PermissionMatrix({ value, onChange }: Props) {
  const { t } = useTranslation();
  const modules = useMemo(() => groupModules(PermissionValues), []);
  const [grantable, setGrantable] = useState<Record<string, boolean>>({});

  useEffect(() => {
    api.get<Grantable[]>("/admin/admins/permissions").then((list) => {
      const map: Record<string, boolean> = {};
      for (const g of list) map[g.permission] = g.grantable;
      setGrantable(map);
    });
  }, []);

  const setLevel = (mod: { read?: Permission; manage?: Permission }, level: Level) => {
    const without = value.filter((p) => p !== mod.read && p !== mod.manage);
    if (level === "read" && mod.read) onChange([...without, mod.read]);
    else if (level === "manage" && mod.manage) onChange([...without, mod.manage]);
    else onChange(without);
  };

  return (
    <div className="flex flex-col gap-3">
      {Array.from(modules.entries()).map(([res, mod]) => {
        const readGrantable = mod.read ? grantable[mod.read] !== false : false;
        const manageGrantable = mod.manage ? grantable[mod.manage] !== false : false;
        const level = currentLevel(value, mod);
        return (
          <div key={res} className="flex items-center gap-4">
            <span className="w-32 font-medium">{t(`permissions.${res}.label`)}</span>
            <RadioGroup
              value={level}
              onChange={(next) => setLevel(mod, next as Level)}
              name={`perm-${res}`}
            >
              <Radio value="none">{t("admin.admins.permission_levels.none")}</Radio>
              <Radio value="read" disabled={!readGrantable || !mod.read}>
                {t("admin.admins.permission_levels.read")}
              </Radio>
              <Radio value="manage" disabled={!manageGrantable || !mod.manage}>
                {t("admin.admins.permission_levels.manage")}
              </Radio>
            </RadioGroup>
          </div>
        );
      })}
    </div>
  );
}
```

```tsx
// frontend/admin/src/pages/AdminFormPage.tsx
import { useEffect, useState } from "react";
import { useNavigate, useParams } from "react-router-dom";
import { useTranslation } from "react-i18next";
import { Button, Input, Select, FormField } from "@shared/components";
import { api } from "@shared/api";
import { useAuth } from "@shared/auth";
import { PermissionMatrix } from "../components/PermissionMatrix";
import type { Permission } from "@shared/types/Permission";
import type { AdminType } from "@shared/types/AdminType";
import type { AdminResponse } from "@shared/types/AdminResponse";

interface Props { mode: "create" | "edit" }

interface FormState {
  username: string;
  email: string;
  name: string;
  password: string;
  admin_type: AdminType;
  permissions: Permission[];
  locale: string;
}

const empty: FormState = {
  username: "", email: "", name: "", password: "",
  admin_type: "Admin", permissions: [], locale: "en",
};

export default function AdminFormPage({ mode }: Props) {
  const { t } = useTranslation();
  const nav = useNavigate();
  const { id } = useParams();
  const { user } = useAuth();
  const [form, setForm] = useState<FormState>(empty);
  const [loading, setLoading] = useState(mode === "edit");

  useEffect(() => {
    if (mode !== "edit" || !id) return;
    api.get<AdminResponse>(`/admin/admins/${id}`).then((a) => {
      setForm({
        username: a.username, email: a.email, name: a.name, password: "",
        admin_type: a.admin_type, permissions: a.permissions as Permission[],
        locale: a.locale,
      });
      setLoading(false);
    });
  }, [id, mode]);

  const allowedTypes: AdminType[] =
    user?.admin_type === "Developer" ? ["Admin", "SuperAdmin"] : ["Admin"];

  const save = async () => {
    if (mode === "create") {
      await api.post("/admin/admins", form);
    } else {
      const { password, username, ...rest } = form;
      const payload: Partial<FormState> = { ...rest };
      if (password) (payload as FormState).password = password;
      await api.put(`/admin/admins/${id}`, payload);
    }
    nav("/admins");
  };

  if (loading) return null;

  return (
    <div className="max-w-2xl flex flex-col gap-4">
      <h1 className="text-2xl font-semibold">
        {t(mode === "create" ? "admin.admins.create" : "admin.admins.edit")}
      </h1>

      {mode === "create" && (
        <FormField label="Username">
          <Input value={form.username} onChange={(v) => setForm({ ...form, username: v })} />
        </FormField>
      )}
      <FormField label="Email">
        <Input value={form.email} onChange={(v) => setForm({ ...form, email: v })} />
      </FormField>
      <FormField label="Name">
        <Input value={form.name} onChange={(v) => setForm({ ...form, name: v })} />
      </FormField>
      <FormField label={mode === "create" ? "Password" : "Password (leave empty to keep)"}>
        <Input
          type="password"
          value={form.password}
          onChange={(v) => setForm({ ...form, password: v })}
        />
      </FormField>
      <FormField label="Admin Type">
        <Select
          value={form.admin_type}
          onChange={(v) => setForm({ ...form, admin_type: v as AdminType })}
          options={allowedTypes.map((t) => ({ value: t, label: t }))}
        />
      </FormField>
      <FormField label={t("admin.admins.fields.permissions")}>
        <PermissionMatrix
          value={form.permissions}
          onChange={(p) => setForm({ ...form, permissions: p })}
        />
      </FormField>

      <div className="flex gap-2">
        <Button onClick={save}>Save</Button>
        <Button unstyled onClick={() => nav("/admins")}>Cancel</Button>
      </div>
    </div>
  );
}
```
